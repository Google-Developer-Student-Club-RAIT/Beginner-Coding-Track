Find the nth term of Harmonic Progression:
Input: A, D(Common difference), N
Output: Nth term

#Example:
Input: A=2, D=4, N=5
Output: 0.055 (Approx)

Contributor Zaid PARKAR

#include <stdio.h>
void main()
{
   int i,n;
   float s=0.0;
   printf("Input the number of terms : ");
   scanf("%d",&n);
   printf("\n\n");
   for(i=1;i<=n;i++)
   {
       if(i<n)
       {
     printf("1/%d + ",i);
     s+=1/(float)i;
       }
     if(i==n)
     {
     printf("1/%d ",i);
     s+=1/(float)i;
     }
     }
        printf("\nSum of Series upto %d terms : %f \n",n,s);
}
